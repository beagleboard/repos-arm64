# These assume the keyring package was built prior to setting up the etc/apt/sources.list

#debian bullseye:
sudo sbuild-shell source:bullseye-arm64-sbuild

apt update ; apt upgrade -yq ; \
echo "deb http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian bullseye main" > /etc/apt/sources.list ; \
echo "deb-src http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian bullseye main" >> /etc/apt/sources.list ; \
echo "deb [trusted=yes] http://repos.rcn-ee.net/debian-arm64 bullseye main" > /etc/apt/sources.list.d/beagle.list ; \
echo "deb-src [trusted=yes] http://repos.rcn-ee.net/debian-arm64 bullseye main" >> /etc/apt/sources.list.d/beagle.list ; \
apt update ; apt upgrade -yq ; apt install -yq bbb.io-keyring ; \
cp -v /etc/bbb.io/templates/apt/sbuild.list /etc/apt/sources.list.d/beagle.list ; \
apt update ; apt upgrade -yq ; apt clean ; echo "done..." ; exit
